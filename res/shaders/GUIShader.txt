##VERTEX

#version 330

layout(location = 0) in vec2 position;

uniform mat4 projMatrix;
uniform mat4 viewMatrix;
uniform mat4 modelMatrix;


void main()
{
	gl_Position = projMatrix * viewMatrix * modelMatrix * vec4(position, 0.0, 1.0);
}

##FRAGMENT

#version 330

uniform vec4 colour;

out vec4 outColour;
void main()
{
	outColour = colour;
}
